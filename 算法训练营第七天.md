```
class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        stack<string> st;
        int n=tokens.size();
        for(int i=0;i<n;i++){
            if(tokens[i]=="+"||tokens[i]=="-"||tokens[i]=="*"||tokens[i]=="/"){
                int num2=stoi(st.top());
                st.pop();
                int num1=stoi(st.top());
                st.pop();
                if(tokens[i]=="+")st.push(to_string(num1+num2));
                else if(tokens[i]=="-")st.push(to_string(num1-num2));
                else if(tokens[i]=="*")st.push(to_string(num1*num2));
                else if(tokens[i]=="/")st.push(to_string(num1/num2));
                else return -1;
            }
            else {
                 st.push(tokens[i]);
               
            }
        }
        if(st.size()==1)return stoi(st.top());
        else return -1;
        
    }
};
```
