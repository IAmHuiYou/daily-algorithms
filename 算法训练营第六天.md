```
    }

        q1.push(x);
        
    }

        while(q1.size()>1){

            q1.pop();
            q2.push(x);

           swap(q1,q2);
```

```
class Solution {

    bool isValid(string s) {
        stack<char> st;
        int n=s.size();
        for(int i=0;i<n;i++){
            if(s[i]=='('||s[i]=='{'||s[i]=='[')st.push(s[i]);
```

```
class Solution {
public:
    string removeDuplicates(string s) {
        stack<char> st;
        int n=s.size();
        for(int i=0;i<n;i++){
            if(st.empty()||st.top()!=s[i])st.push(s[i]);
            else
            st.pop();
        }
        n=st.size();
        string result;
        for(int i=0;i<n;i++){
            result+=st.top();
            st.pop();
        }
        reverse(result.begin(),result.end());
        return result;
        
    }
};
```
