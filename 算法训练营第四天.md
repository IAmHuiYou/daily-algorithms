```
    void reverseString(vector<char>& s) {
        int n=s.size();
        int i=0,j=n-1;
        while(i<j){
            swap(s[i++],s[j--]);
        }
        
    }
};
```

## 我的代码

```
class Solution {
public:

     void swaps(string &s,int start,int end){
        while(start<end){
            swap(s[start++],s[end--]);
        }
    }

    string reverseStr(string s, int k) {
        int n=s.size();
        for(int i=0;i<n;i+=2*k){
            int start=i;
            int end=min(start+k-1,n-1);
            swaps(s,start,end);
        }
        return s;
        
    }
   
};
```
